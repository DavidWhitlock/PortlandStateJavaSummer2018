001package java7;
002
003import com.sandwich.koan.Koan;
004
005import static com.sandwich.koan.constant.KoanConstants.__;
006import static com.sandwich.util.Assert.assertEquals;
007
008public class AboutStringsInSwitch {
009
010    @Koan
011    public void stringsInSwitchStatement() {
012        String[] animals = {"Dog", "Cat", "Tiger", "Elephant", "Zebra"};
013        String dangerous = null;
014        String notDangerous = null;
015        for (String animal : animals) {
016            switch (animal) {
017                case "Tiger":
018                    dangerous = animal;
019                case "Dog":
020                case "Cat":
021                case "Elephant":
022                case "Zebra":
023                    notDangerous = animal;
024            }
025        }
026        assertEquals(notDangerous, __);
027        assertEquals(dangerous, __);
028    }
029
030}

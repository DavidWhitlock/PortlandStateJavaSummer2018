001package java8;
002
003import com.sandwich.koan.Koan;
004
005import java.time.LocalTime;
006import java.time.temporal.ChronoUnit;
007
008import static com.sandwich.koan.constant.KoanConstants.__;
009import static com.sandwich.util.Assert.assertEquals;
010
011public class AboutLocalTime {
012
013    @Koan
014    public void localTime() {
015        LocalTime t1 = LocalTime.of(7, 30);
016        assertEquals(t1, LocalTime.parse(__));
017    }
018
019    @Koan
020    public void localTimeMinus() {
021        LocalTime t1 = LocalTime.parse("10:30");
022        LocalTime t2 = t1.minus(2, ChronoUnit.HOURS);
023        assertEquals(t2, LocalTime.parse(__));
024    }
025
026}

001package java8;
002
003import com.sandwich.koan.Koan;
004
005import java.util.Base64;
006import java.io.UnsupportedEncodingException;
007
008import static com.sandwich.koan.constant.KoanConstants.__;
009import static com.sandwich.util.Assert.assertEquals;
010
011public class AboutBase64 {
012
013    private final String plainText = "lorem ipsum";
014    private final String encodedText = "bG9yZW0gaXBzdW0=";
015
016    @Koan
017    public void base64Encoding() {
018        try {
019            // Encode the plainText
020            // This uses the basic Base64 encoding scheme but there are corresponding
021            // getMimeEncoder and getUrlEncoder methods available if you require a
022            // different format/Base64 Alphabet 
023            assertEquals(encodedText, Base64.getEncoder().encodeToString(__.getBytes("utf-8")));
024        } catch (UnsupportedEncodingException ex) {}
025    }
026
027    @Koan
028    public void base64Decoding() {
029        // Decode the Base64 encodedText
030        // This uses the basic Base64 decoding scheme but there are corresponding
031        // getMimeDecoder and getUrlDecoder methods available if you require a
032        // different format/Base64 Alphabet
033        byte[] decodedBytes = Base64.getDecoder().decode(__);
034        try {
035            String decodedText = new String(decodedBytes, "utf-8");
036            assertEquals(plainText, decodedText);
037        } catch (UnsupportedEncodingException ex) {}
038    }
039
040}

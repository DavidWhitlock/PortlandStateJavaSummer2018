001package edu.pdx.cs410J.whitlock;
002
003import org.junit.Test;
004
005import java.util.ArrayList;
006
007import static org.hamcrest.CoreMatchers.containsString;
008import static org.hamcrest.MatcherAssert.assertThat;
009import static org.hamcrest.core.IsEqual.equalTo;
010
011/**
012 * Unit tests for the Student class.  In addition to the JUnit annotations,
013 * they also make use of the <a href="http://hamcrest.org/JavaHamcrest/">hamcrest</a>
014 * matchers for more readable assertion statements.
015 */
016public class StudentTest
017{
018
019  @Test
020  public void studentNamedPatIsNamedPat() {
021    String name = "Pat";
022    var pat = new Student(name, new ArrayList<>(), 0.0, "Doesn't matter");
023    assertThat(pat.getName(), equalTo(name));
024  }
025
026  @Test(expected = IllegalArgumentException.class)
027  public void gpaMustBeLessThanOrEqualToFourPointZero() {
028    createStudentWithGpa(4.1);
029  }
030
031  @Test
032  public void aFourPointZeroGpaIsPossible() {
033    createStudentWithGpa(4.0);
034  }
035
036  @Test(expected = IllegalArgumentException.class)
037  public void gpaMustBeGreaterThanZero() {
038    createStudentWithGpa(-1.0);
039  }
040
041  @Test
042  public void aZeroGpaIsPossible() {
043    createStudentWithGpa(0.0);
044  }
045
046  private void createStudentWithGpa(double gpa) {
047    new Student("??", new ArrayList<>(), gpa, "Doesn't matter");
048  }
049
050  @Test
051  public void javaClassShowsUpInToString() {
052    String className = "Java";
053    var classes = new ArrayList<String>();
054    classes.add(className);
055
056    //When:
057    Student student = new Student("??", classes, 3.8, "Doesn't matter");
058
059    //Then:
060    assertThat(student.toString(), containsString(className));
061  }
062
063}

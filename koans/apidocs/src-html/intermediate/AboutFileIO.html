001package intermediate;
002
003import com.sandwich.koan.Koan;
004
005import java.io.*;
006import java.util.logging.Logger;
007
008import static com.sandwich.koan.constant.KoanConstants.__;
009import static com.sandwich.util.Assert.assertEquals;
010
011public class AboutFileIO {
012
013    @Koan
014    public void fileObjectDoesntCreateFile() {
015        File f = new File("foo.txt");
016        assertEquals(f.exists(), __);
017    }
018
019    @Koan
020    public void fileCreationAndDeletion() throws IOException {
021        File f = new File("foo.txt");
022        f.createNewFile();
023        assertEquals(f.exists(), __);
024        f.delete();
025        assertEquals(f.exists(), __);
026    }
027
028    @Koan
029    public void basicFileWritingAndReading() throws IOException {
030        File file = new File("file.txt");
031        FileWriter fw = new FileWriter(file);
032        fw.write("First line\nSecond line");
033        fw.flush();
034        fw.close();
035
036        char[] in = new char[50];
037        int size = 0;
038        FileReader fr = new FileReader(file);
039        size = fr.read(in);
040        // No flush necessary!
041        fr.close();
042        assertEquals(size, __);
043        String expected = new String(in);
044        assertEquals(expected.length(), __);
045        assertEquals(expected, __);
046        file.delete();
047    }
048
049    @Koan
050    public void betterFileWritingAndReading() throws IOException {
051        File file = new File("file.txt");
052        file.deleteOnExit();
053        FileWriter fw = new FileWriter(file);
054        PrintWriter pw = new PrintWriter(fw);
055        pw.println("First line");
056        pw.println("Second line");
057        pw.close();
058
059        FileReader fr = new FileReader(file);
060        BufferedReader br = null;
061        try {
062            br = new BufferedReader(fr);
063            assertEquals(br.readLine(), __); // first line
064            assertEquals(br.readLine(), __); // second line
065            assertEquals(br.readLine(), __); // what now?
066        } finally {
067            // anytime you open access to a file, you should close it or you may
068            // lock it from other processes (ie frustrate people)
069            closeStream(br);
070        }
071    }
072
073    private void closeStream(BufferedReader br) {
074        if (br != null) {
075            try {
076                br.close();
077            } catch (IOException x) {
078                Logger.getAnonymousLogger().severe("Unable to close reader.");
079            }
080        }
081    }
082
083    @Koan
084    public void directChainingForReadingAndWriting() throws IOException {
085        File file = new File("file.txt");
086        PrintWriter pw = new PrintWriter(file);
087        pw.println("1. line");
088        pw.println("2. line");
089        pw.close();
090
091        StringBuffer sb = new StringBuffer();
092        // Add the loop to go through the file line by line and add the line
093        // to the StringBuffer
094        assertEquals(sb.toString(), "1. line\n2. line");
095    }
096}

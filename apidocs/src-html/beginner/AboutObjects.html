001package beginner;
002
003import com.sandwich.koan.Koan;
004
005import java.text.MessageFormat;
006import java.util.ArrayList;
007import java.util.List;
008
009import static com.sandwich.koan.constant.KoanConstants.__;
010import static com.sandwich.util.Assert.assertEquals;
011
012public class AboutObjects {
013
014    @Koan
015    public void newObjectInstancesCanBeCreatedDirectly() {
016        assertEquals(new Object() instanceof Object, __);
017    }
018
019    @Koan
020    public void allClassesInheritFromObject() {
021        class Foo {
022        }
023
024        Class<?>[] ancestors = getAncestors(new Foo());
025        assertEquals(ancestors[0], __);
026        assertEquals(ancestors[1], __);
027    }
028
029    @Koan
030    public void objectToString() {
031        Object object = new Object();
032        // TODO: Why is it best practice to ALWAYS override toString?
033        String expectedToString = MessageFormat.format("{0}@{1}", Object.class.getName(), Integer.toHexString(object.hashCode()));
034        assertEquals(expectedToString, __); // hint: object.toString()
035    }
036
037    @Koan
038    public void toStringConcatenates() {
039        final String string = "ha";
040        Object object = new Object() {
041            @Override
042            public String toString() {
043                return string;
044            }
045        };
046        assertEquals(string + object, __);
047    }
048
049    @Koan
050    public void toStringIsTestedForNullWhenInvokedImplicitly() {
051        String string = "string";
052        assertEquals(string + null, __);
053    }
054
055    private Class<?>[] getAncestors(Object object) {
056        List<Class<?>> ancestors = new ArrayList<Class<?>>();
057        Class<?> clazz = object.getClass();
058        while (clazz != null) {
059            ancestors.add(clazz);
060            clazz = clazz.getSuperclass();
061        }
062        return ancestors.toArray(new Class[]{});
063    }
064
065}

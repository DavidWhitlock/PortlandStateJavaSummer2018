001package beginner;
002
003import com.sandwich.koan.Koan;
004
005import static com.sandwich.koan.constant.KoanConstants.__;
006import static com.sandwich.util.Assert.*;
007
008public class AboutAssertions {
009
010    @Koan
011    public void assertBooleanTrue() {
012        // there are two possibilities, true or false, what would it be here?
013        assertTrue(true);
014    }
015
016    @Koan
017    public void assertBooleanFalse() {
018        assertFalse(false);
019    }
020
021    @Koan
022    public void assertNullObject() {
023        // reference to the object can be null, a magic keyword, null, which means
024        // that there is nothing there
025        assertNull(null);
026    }
027
028    @Koan
029    public void assertNullObjectReference() {
030        Object someObject = null;
031        assertNull(someObject);
032    }
033
034    @Koan
035    public void assertNotNullObject() {
036        // but what when there should not be a null value?
037        assertNotNull(1.0f);
038    }
039
040    @Koan
041    public void assertEqualsUsingExpression() {
042        assertTrue("Hello World!".equals("Hello World!"));
043    }
044
045    @Koan
046    public void assertEqualsWithAFewExpressions() {
047        assertEquals("Hello World!", "Hello World!");
048        assertEquals(1, 1);
049        assertEquals(2 + 2, 4);
050        assertEquals(2 * 3, 6);
051        assertEquals(3 - 8, -5);
052        assertEquals(10 / 2, 5);
053    }
054
055    @Koan
056    public void assertEqualsWithDescriptiveMessage() {
057        // Generally, when using an assertXXX methods, expectation is on the
058        // left and it is best practice to use a String for the first arg
059        // indication what has failed
060        assertEquals("The answer to 'life the universe and everything' should be 42", 42, 6*7);
061    }
062
063    @Koan
064    public void assertSameInstance() {
065        Integer original = new Integer(1);
066        Integer same = original;
067        Integer different = new Integer(1);
068        // These are both equal to the original...
069        assertEquals(original, same);
070        assertEquals(original, different);
071        // ...but only one refers to the same instance as the original.
072        assertSame(original, same);
073    }
074
075    @Koan
076    public void assertNotSameInstance() {
077        Integer original = new Integer(1);
078        Integer same = original;
079        Integer different = new Integer(1);
080        // These are both equal to the original...
081        assertEquals(original, same);
082        assertEquals(original, different);
083        // ...but only one of them refers to a different instance.
084        assertNotSame(original, different);  // We want equal, but _not_ the same.
085    }
086}
